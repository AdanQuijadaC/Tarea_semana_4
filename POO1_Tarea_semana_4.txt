import java.util.Random;
public class Persona
{

    // Declaracion de atributos de la clase
    private String nombre;
    private String apellido;
    private int edad;
    private String RUT = "";
    private int peso;
    private double altura;
    private char sexo = 'H';

    // metodo main
    public static void main(String[] args)
    {
        System.out.print("Llamada al metodo main");

    }
    // Constructores
    public Persona()
    {
        nombre = "";
        apellido = "";
    }

    public Persona(String nombre, int edad, char sexo)
    {
        this.nombre = nombre;
        this.edad = edad;
        this.sexo = sexo;
    }

    public Persona(String nombre, int edad, char sexo, String RUT, int peso, double altura, String apellido)
    {
        this.nombre = nombre;
        this.edad = edad;
        this.RUT = RUT;
        this.peso = peso;
        this.altura = altura;
        this.apellido = apellido;
    }

    // Metodos
    public int calcularIMC(int peso, double altura) // Calcular IMC
    {
        double valorIMC = peso / (altura * altura);
        int valorIMC_int = (int) valorIMC;

        if(valorIMC_int<20)
        {
            return -1;
        }else if(valorIMC_int >= 20 & valorIMC_int <=25)
        {
            return 0; 
        }else
        {
            return 1;
        }

    }

    // Metodo comprobar sexo
    public char comprobarSexo(char sexo)
    {
        if(sexo=='H')
        {

            return 'H';
        }else
        {
            sexo = 'H';
            return sexo;
        }
    }

    // Metodo generar RUT
    public String generarRUT()
    {

        Random rnd = new Random(); // Llamada a la funcion Random rnd
        int numeroAleatorio_int = (int)(rnd.nextDouble()*100000000+1);

        String numeroString = String.valueOf(numeroAleatorio_int); // Conversion de int a String

        String barra = "-";
        int numeroExtension = (int)(rnd.nextDouble()*8+1); 

        String Extension_String = String.valueOf(numeroExtension); // Conversion de int a String

        String conbinacion_rut = numeroString+barra+Extension_String; // Suma de cadenas

        return conbinacion_rut;
    }
    /* Metodos set  */
    public void setNombre(String newNombre)
    {
        nombre = newNombre;
    }

    public void setApellido(String newApellido)
    {
        apellido = newApellido;
    }

    public void setEdad(int newEdad)
    {
        edad = newEdad;
    }

    public void setPeso(int newPeso)
    {
        peso = newPeso;
    }

    public void setAltura(double newAltura)
    {
        altura = newAltura;
    }

    public void setSexo(char newSexo)
    {
        sexo = newSexo;
    }

}
